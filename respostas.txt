5- SELECT c.nome, COUNT(*) AS total_contratos
FROM cliente c
JOIN contrato ct ON c.id_cliente = ct.id_cliente
GROUP BY c.nome
ORDER BY total_contratos DESC
LIMIT 1;

6- SELECT id_veiculo, COUNT(*) AS total
FROM manutencao
GROUP BY id_veiculo
HAVING COUNT(*) > 1;

7- SELECT id_veiculo, SUM(valor) AS total_manutencao
FROM manutencao
GROUP BY id_veiculo;

8- SELECT DISTINCT c.nome
FROM cliente c

9- SELECT DISTINCT c.nome
FROM cliente c

10- SELECT id_contrato, SUM(valor_multa) AS total_multas
FROM multa
GROUP BY id_contrato;

11- SELECT c.nome, SUM(m.valor_multa) AS total
FROM cliente c
JOIN contrato ct ON c.id_cliente = ct.id_cliente
JOIN multa m ON ct.id_contrato = m.id_contrato
GROUP BY c.nome
ORDER BY total DESC
LIMIT 1;

12- SELECT id_empresa, COUNT(*) AS total_veiculos
FROM veiculo

13- SELECT e.nome, v.valor
FROM empresa e
JOIN veiculo v ON e.id_empresa = v.id_empresa

14- SELECT c.nome, ct.valor, ct.quilometragem_inicial

15- SELECT c.nome
FROM cliente c

17- SELECT c.nome,
       d.km_final - ct.quilometragem_inicial AS km_percorrido
FROM cliente c
JOIN contrato ct ON c.id_cliente = ct.id_cliente
JOIN devolucao d ON ct.id_contrato = d.id_contrato;

18- SELECT ct.*
FROM contrato ct
JOIN veiculo v ON ct.id_veiculo = v.id_veiculo

19- SELECT c.nome, c.data_nascimento
